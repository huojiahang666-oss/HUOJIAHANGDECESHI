#!/usr/bin/env python3 # -*- coding: utf-8 -*- import tkinter as tk from tkinter import simpledialog, scrolledtext import asyncio import websockets import threading # 修改为你的 ngrok 公网 wss 地址 NGROK_URL = "wss://ducally-chemiluminescent-zenaida.ngrok-free.dev" class ChatClient: def __init__(self, root): self.root = root self.root.title("聊天室") self.loop = asyncio.new_event_loop() threading.Thread(target=self.loop.run_forever, daemon=True).start() # 聊天记录框 self.text_area = scrolledtext.ScrolledText(root, state='disabled', width=50, height=20) self.text_area.pack(padx=10, pady=10) # 输入框 self.input_field = tk.Entry(root, width=40) self.input_field.pack(side=tk.LEFT, padx=(10,0), pady=(0,10)) send_button = tk.Button(root, text="发送", command=self.on_send) send_button.pack(side=tk.LEFT, padx=(5,10), pady=(0,10)) # 用户名 self.username = simpledialog.askstring("用户名", "请输入用户名:", parent=self.root) if not self.username: self.username = "匿名" self.websocket = None # 连接服务器 asyncio.run_coroutine_threadsafe(self.connect(), self.loop) async def connect(self): try: self.websocket = await websockets.connect(NGROK_URL) asyncio.create_task(self.receive_messages()) except Exception as e: self.append_text(f"[错误] 无法连接服务器: {e}") async def receive_messages(self): try: async for message in self.websocket: self.append_text(message) except Exception as e: self.append_text(f"[错误] 连接断开: {e}") def append_text(self, message): self.text_area.configure(state='normal') self.text_area.insert(tk.END, message + "\n") self.text_area.configure(state='disabled') self.text_area.see(tk.END) async def send_message_async(self, message): if self.websocket: try: await self.websocket.send(f"{self.username}: {message}") except Exception as e: self.append_text(f"[错误] 发送失败: {e}") def on_send(self): message = self.input_field.get().strip() if message: self.append_text(f"{self.username}: {message}") # 显示自己消息 asyncio.run_coroutine_threadsafe(self.send_message_async(message), self.loop) self.input_field.delete(0, tk.END) if __name__ == "__main__": root = tk.Tk() ChatClient(root) root.mainloop()
